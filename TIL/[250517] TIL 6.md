```python
def solution(s):
    answer = ""
    cnt = 0
    for i in range(len(s)):
        if s[i] == " ":
            cnt = 0
            answer += s[i]
        else:
            if cnt % 2 == 0:
                answer += s[i].upper()
            else:
                answer += s[i].lower()
            cnt += 1
    return answer
```

```python
def solution(number):
    answer = 0
    for i in range(0, len(number) - 2):
        for j in range(i + 1, len(number) - 1):
            for k in range(j + 1, len(number)):
                if number[i] + number[j] + number[k] == 0:
                    answer += 1
    return answer
```

```python
def solution(t, p):
    answer = 0
    num_list = [t[i:len(p) + i] for i in range(len(t) - len(p) + 1)]
    for num in num_list:
        if int(p) >= int(num):
```

```python
def solution(sizes):
    tmp = []
    width = 0
    length = 0
    for i in range(len(sizes)):
        tmp.append(sorted(sizes[i]))
        if width < tmp[i][0]:
            width = tmp[i][0]
        if length < tmp[i][1]:
            length = tmp[i][1]
    return width * length
```

```python
def solution(s, n):
    answer = ''
    for ch in s:
        # 공백 처리
        if ch == " ":
            answer += " "
            continue
        else:
            # 대문자 중에서
            if ch >= 'A' and ch <= 'Z':
                # Z를 넘어가면
                if chr(ord(ch) + n) > 'Z':
                    answer += chr(65 + (ord(ch) + n) - ord('Z') - 1)
                # 안 넘어가면
                else:
                    answer += chr(ord(ch) + n)
            # 소문자 중에서
            elif ch >= 'a' and ch <='z':
                # z를 넘어가면
                if chr(ord(ch) + n) > 'z':
                    answer += chr(97 + (ord(ch) + n) - ord('z') - 1)
                # 안 넘어가면
                else:
                    answer += chr(ord(ch) + n)
    return answer
```

```python
def solution(s):
```

```python
def solution(strings, n):
    return sorted(strings, key=lambda x: (x[n], x))
```
